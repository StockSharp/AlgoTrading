exit_hour = input.int(23, title="Exit Hour (UTC)", minval=0, maxval=23)

// Define current hour and minute
current_hour = hour
current_minute = minute

// Trading logic based on user-defined times
enterTrade = (current_hour == entry_hour) and (current_minute == 0)
