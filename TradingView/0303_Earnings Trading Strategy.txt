//@version=5
strategy("Earnings Trading Strategy", overlay=true)

// Input for trade direction
direction = input.string("Long", "Direction", options=["Long", "Short"])

// Inputs for earnings dates (entry and exit trading days)
entry_date1 = input.time(timestamp("2025-03-13"), "Entry Date 1")
exit_date1 = input.time(timestamp("2025-03-17"), "Exit Date 1")
entry_date2 = input.time(timestamp("2025-06-12"), "Entry Date 2")
exit_date2 = input.time(timestamp("2025-06-16"), "Exit Date 2")

// Define quantity
qty = 100 // Fixed 100 shares per trade

// Get current bar's timestamp (daily bars assumed)
current_time = time

// Entry conditions
enter_long = (current_time == entry_date1) or (current_time == entry_date2)
enter_short = (current_time == entry_date1) or (current_time == entry_date2)

// Exit conditions
exit = (current_time == exit_date1) or (current_time == exit_date2)

// Execute trades based on direction
if (direction == "Long")
    if (enter_long)
        strategy.entry("Long", strategy.long, qty)
    if (exit)
        strategy.close("Long")
else if (direction == "Short")
    if (enter_short)
        strategy.entry("Short", strategy.short, qty)
    if (exit)
        strategy.close("Short")
