strategy("Cycle Biologique Strategy", overlay=true)

// Paramètres pour ajuster la période du cycle
cycle_length = input.int(30, title="Longueur du cycle", minval=1)
amplitude = input.float(1.0, title="Amplitude du cycle", step=0.1)
offset = input.int(0, title="Décalage du cycle", minval=-360, maxval=360)

// Calcul du cycle périodique avec une fonction sinusoïdale
cycle_value = amplitude * math.sin((2 * math.pi * (bar_index + offset)) / cycle_length)

// Indicateurs d'achat et de vente en fonction du cycle
buy_signal = ta.crossover(cycle_value, 0)
sell_signal = ta.crossunder(cycle_value, 0)

// Exécution des ordres en fonction des signaux
if (buy_signal)
    strategy.entry("Long Entry", strategy.long)

if (sell_signal)
    strategy.close("Long Entry")

// Afficher le cycle périodique sur le graphique
plot(cycle_value, color=color.blue, title="Cycle Biologique")
