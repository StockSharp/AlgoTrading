//@version=5
strategy("EMA Crossover Strategy with Shift", overlay=true)

// Input parameters for EMAs
shortEmaLength = input.int(9, title="Short EMA Length")
longEmaLength = input.int(21, title="Long EMA Length")
shiftAmount = input.int(-50, title="Shift Amount")

// Calculate EMAs and apply shift
shortEma = ta.ema(close, shortEmaLength) + shiftAmount
longEma = ta.ema(close, longEmaLength) + shiftAmount

// Plot EMAs on the chart
plot(shortEma, color=color.blue, title="Short EMA (Shifted)")
plot(longEma, color=color.red, title="Long EMA (Shifted)")

// Generate buy and sell signals
longCondition = ta.crossover(shortEma, longEma)
shortCondition = ta.crossunder(shortEma, longEma)

// Execute buy and sell orders
if (longCondition)
    strategy.entry("Long", strategy.long)

if (shortCondition)
    strategy.entry("Short", strategy.short)
