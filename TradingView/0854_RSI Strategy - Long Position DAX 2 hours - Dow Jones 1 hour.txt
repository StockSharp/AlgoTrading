//@version=5
strategy("Stratégie RSI Long", overlay=true)

// Calcul du RSI
rsiLength = input.int(14, title="Longueur du RSI")
rsiValue = ta.rsi(close, rsiLength)

// Conditions d'entrée et de sortie
enterLong = ta.crossover(rsiValue, 35)
exitLong = rsiValue > 55 or ta.crossunder(rsiValue, 30)

// Exécution de la stratégie
if (enterLong)
    strategy.entry("Achat", strategy.long)

if (exitLong)
    strategy.close("Achat")
