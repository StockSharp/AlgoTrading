strategy("CE XAU/USDT Strategy", overlay=true, default_qty_type=strategy.percent_of_equity, default_qty_value=100)

// Условный индикатор CE (замена сигнала Buy/Sell, если нет доступа к настоящему коду)
ceBuy  = ta.crossover(close, ta.sma(close, 14))  // Имитация сигнала Buy
ceSell = ta.crossunder(close, ta.sma(close, 14)) // Имитация сигнала Sell

if (ceBuy)
