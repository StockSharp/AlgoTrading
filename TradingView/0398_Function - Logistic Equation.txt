//@version=4
study(title='Function - Logistic Equation')

f_logde(_pop, _limit)=>1. * _pop * (1. - (_pop / _limit))

int length = input(14)

float pop = change(close, length)

float limit = highest(length)-lowest(length)

logde_neg = f_logde(pop, limit)
logde_pos = -f_logde(-pop, limit)

plot(series=logde_pos, title='logistic eq', color=color.green)
plot(series=logde_neg, title='logistic eq', color=color.maroon)
hline(0)

float upper = limit*0.5
float lower = -upper
rp = plot(series=upper, title='R+', color=color.gray)
rm = plot(series=lower, title='R-', color=color.gray)
fill(rp, rm, color=color.gray, transp=70)


color bgc = logde_neg < lower ? color.new(color.maroon, 0) : (logde_pos > upper ? color.new(color.green, 0) : color.new(color.silver, 0))
barcolor(bgc)
